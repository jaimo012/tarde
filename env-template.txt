# 로컬 개발환경용 환경변수 템플릿

# .env 파일은 .gitignore에 포함되어 GitHub에 업로드되지 않습니다

# DART API 설정
DART_API_KEY=your_dart_api_key_here

# 구글 스프레드시트 설정
